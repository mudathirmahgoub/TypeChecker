; This example illustrates basic arithmetic and
; uninterpreted functions

(declare-fun a () Int)
(declare-fun b () Int)
(assert (= (mod a 2) 0))
(assert (<= (+ a b) -3))
(assert (<= (-(* 2 b) a) -7))
(assert (not (= (+ a (* 3 b)) 15)))
