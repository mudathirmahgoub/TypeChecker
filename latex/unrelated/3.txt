; This example illustrates basic arithmetic and
; uninterpreted functions

(declare-fun a () Int)
(declare-fun b () Int)
(assert (not (= (mod a 2) 0)))
(assert (> b -6))
(assert (not (= (+ (* 2 a) b) 12)))
(assert (= a 1))
(assert (= b 0))
(check-sat)
(get-model)
