; This example illustrates basic arithmetic and
; uninterpreted functions

(declare-fun a () Int)
(declare-fun b () Int)
(assert (= (mod a 2) 0))
(assert (> (+ a b) -3))
(assert (<= 6 a))
(assert (not (= (+ a (* 3 b)) 15)))
(check-sat)
(get-model)

sat (model (define-fun b () Int 0) (define-fun a () Int 6) )
